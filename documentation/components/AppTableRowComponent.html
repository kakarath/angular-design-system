import { Component, Input, OnInit, EventEmitter, Output } from '@angular/core';

@Component({
  selector: '[app-table-row]',
  templateUrl: './table-row.component.html',
  styleUrls: ['./table-row.component.css']
})

export class AppTableRowComponent implements OnInit {
    @Input() dataRow;
    @Input() headers;
    @Input() index;
    @Input() rowHeader;
    @Input() searchText;
    @Input() highlightSearch;
    @Output() onCallBack = new EventEmitter<any>();

    shadowArray = [];

    ngOnInit() {
        Object.keys(this.headers).map((key) => {
            this.shadowArray.push(
                {
                    key: this.headers[key].header.prop,
                    value: this.dataRow[this.headers[key].header.prop],
                    header: this.headers[key].header.value
                }
            );
        });
    }

    callBack(e) {
        this.onCallBack.emit(e);
    }
}
<ng-container *ngFor="let data of shadowArray; let i = index">
    <th *ngIf="data.key === 'row_Header'" 
        scope="row" 
        app-table-data 
            [dataClass]="'rowHeader'" 
            [data] = "data" 
            [index]="index + '_' + i"
            [searchText]="searchText"
            [highlightSearch]="highlightSearch"
            (callBack)="callBack($event)"></th>
    
    <td *ngIf="data.key !== 'row_Header'" 
        [attr.data-column-header]="data.header" 
        app-table-data 
            [dataClass]="'columnCell'" 
            [data] = "data" 
            [index]="index + '_' + i"
            [searchText]="searchText"
            [highlightSearch]="highlightSearch"
            (callBack)="callBack($event)"></td>
</ng-container>
