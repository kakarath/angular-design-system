import {
    Component,
    Input,
    Output,
    EventEmitter
} from '@angular/core';

@Component({
  selector: 'app-table-modal',
  templateUrl: './table-modal.component.html',
  styleUrls: ['./table-modal.component.css']
})
export class AppTableModalComponent {

    @Input() modalTitle;
    @Input() modalData: Array<any>;
    @Input() modalCount: number;

    @Output() closeModal = new EventEmitter<any>();

    onClose() {
        this.closeModal.emit(true);
    }

}
<ng-template #bodyTemplate let-modalBody="modalBody">
    <ng-container *ngFor="let data of modalBody.modalData">
        <h2 *ngIf="data.heading" [ngClass]="'ds-h2 ' + data.class">
            {{ data.heading }}
        </h2>
        <p *ngIf="data.paragraph" [ngClass]="'ds-text ' + data.class">
            {{ data.paragraph }}
        </p>
        <h3 *ngIf="data.heading2" [ngClass]="'ds-h4 ' + data.class">
            {{ data.heading2 }}
        </h3>
        <hr *ngIf="data.hline" [ngClass]="data.class"/>
    </ng-container>
</ng-template>
