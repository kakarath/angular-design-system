import { Component, Input, EventEmitter, Output } from '@angular/core';

@Component({
    selector: 'app-drop-down',
    templateUrl: './drop-down.component.html',
    styleUrls: ['./drop-down.component.css']
})
export class AppDropDownComponent {
    @Input() options;
    @Input() labelName: string;
    @Input() selectClass: string;
    @Input() id = 1;
    @Input() defaultSelected = 0;
    @Input() hintMessage: string;
    @Input() errorMessage: string;
    @Input() error: boolean;
    @Input() dataAutoId: string;
    @Output() selectedOption = new EventEmitter<any>();

    selectOption(e) {
        this.selectedOption.emit(e.target.value);
    }
}
<label
    *ngIf="labelName" 
    class="ds-c-label ds-u-margin-top--0 ds-u-sans" 
    for="{{'dropdown-' + id}}">
        {{ labelName }}
    <span *ngIf="hintMessage" class="ds-c-field__hint ds-c-field__hint--inverse">{{ hintMessage }}</span>
    <span *ngIf="error" class="ds-c-field__hint ds-u-color--error" id="undefined-message" role="alert">{{ errorMessage }}</span>
</label>
<select 
        id="{{'dropdown-' + id}}" 
        class="ds-c-field {{selectClass}}"
        [ngClass]="{'ds-c-field--error': error}"
        name="options"
        (change)="selectOption($event)"
        attr.data-auto-id="{{dataAutoId}}">
    <ng-container *ngFor="let option of options; let i = index">
        <option *ngIf="i === defaultSelected; else notSelected" value="{{ option.value }}" selected>
            {{ option.content }}
        </option>
        <ng-template #notSelected>
            <option value="{{ option.value }}">
                {{ option.content }}
            </option>
        </ng-template>
    </ng-container>
</select>
